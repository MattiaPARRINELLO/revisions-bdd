OpérationCode
ConnexionMongoClient("mongodb://localhost:27017/")
Choisir une baseclient["nom_base"]
Choisir une collectiondb["nom_collection"]
Insérer (Create)
# Insérer un document
db["players"].insert_one({
    "username": "mattia",
    "score": 0,
    "monney": 100
})

# Insérer plusieurs
db["players"].insert_many([
    {"username": "alice", "score": 50},
    {"username": "bob", "score": 30},
])
Lire (Read)
# Trouver un seul document
player = db["players"].find_one(
    {"username": "mattia"}
)
# Retourne un dict ou None

# Trouver plusieurs documents
all_players = list(db["players"].find())

# Avec projection (choisir les champs)
players = list(db["players"].find(
    {},                             # filtre vide = tous
    {"_id": 0, "username": 1, "score": 1}  # 1 = inclure, 0 = exclure
))
Mettre à jour (Update)
# Incrémenter une valeur
db["players"].update_one(
    {"username": "mattia"},
    {"$inc": {"monney": 50}}
)

# Modifier un champ
db["players"].update_one(
    {"username": "mattia"},
    {"$set": {"level": 5}}
)
OpérateurEffet
$incAjoute une valeur numérique
$setModifie/ajoute un champ
$pushAjoute à un tableau
Supprimer (Delete)
# Supprimer un document
db["players"].delete_one(
    {"username": "mattia"}
)

# Supprimer plusieurs
db["players"].delete_many(
    {"score": {"$lt": 10}}  # score < 10
)

delete_many({}) supprime TOUT

Trier, Limiter, Compter
# Trier (1 = croissant, -1 = décroissant)
top = list(db["players"].find()
    .sort("score", -1)
    .limit(3))

# Compter
count = db["players"].count_documents({})

from pymongo import MongoClient


"""Base de données"""

def open_db():
    client = MongoClient("mongodb://localhost:27017/")
    return client["Game"]


def get_collection(db, name):
    return db[name]


"""Recherche"""

def get_player(db, username):
    players = get_collection(db, "players")
    return players.find_one({"username": username})

Classes
class Character:
    def __init__(self, defense, atk, name, max_pv):
        self.defense = defense
        self.atk = atk
        self.name = name
        self.max_pv = max_pv
        self.current_pv = max_pv
        self.effets = Stack()

    def attack(self, cible):
        dmg = self.atk - cible.defense
        cible.current_pv -= dmg if dmg > 0 else 1

    def is_dead(self):
        return self.current_pv <= 0

Stack maison
class Stack:
    def __init__(self):
        self.stack = []

    def empiler(self, valeur):   # push
        self.stack.append(valeur)

    def depiler(self):           # pop
        return self.stack.pop()

    def is_empty(self):
        return len(self.stack) == 0

Principes généraux
  • Une fonction = une responsabilité
  • Paramètres explicites — pas de globales
  • Early return — éviter les indentations profondes
  • Dictionnaire pour les menus
  • while pour les validations de saisie
  • Conditions explicites
# Early return
def team_is_dead(team):
    for c in team:
        if not c.is_dead():
            return False
    return True
